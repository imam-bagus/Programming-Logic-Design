Declare int bookPurchased
Declare int point

Output "Number of books purchased :"
Input bookPurchased

If bookPurchased < 1 Then
    Set point = 0
Else
    If bookPurchased == 1 Then
        Set point = 5   
    Else
        If bookPurchased == 2 Then
            Set point = 15   
        Else
            If bookPurchased == 3 Then
                Set point = 30
            Else
                    Set point = 60    
            EndIf
        EndIf
    EndIf
EndIf

Output "Points awarded :", point
