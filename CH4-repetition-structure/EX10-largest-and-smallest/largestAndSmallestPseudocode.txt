Declare int numbers
Declare int largest
Declare int smallest
Declare int flag = 1

Output "Enter a number : "
Input numbers

While numbers != -99 

    If flag == 1 Then
       Set largest = numbers
       Set smallest = numbers
    EndIf

    If numbers > largest Then
        largest = numbers
    Else
        If numbers < smallest Then
        Set smallest = numbers
        EndIf
    EndIf

    Output "Enter a number : "
    Input numbers

    Set flag = flag + 1
EndWhile

Output "Largest number :", largest
Output "Smallest number :", smallest
