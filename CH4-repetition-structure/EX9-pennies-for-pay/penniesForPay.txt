Declare Integer day
Declare Integer days
Declare Real salary
Declare Real totalPay

Output "Enter a days :"
Input days

Output "_____________________"
Output "|  Day  |   salary  |"

Set day = 1
For day To days Then

  Set salary = day / 100
  Set totalPay = totalPay + salary
  
  Output "|   ", day, "   |   $", salary, "   |"

  Set day = day + 1
EndFor

Output "|  TOTAL  |   $", totalPay, " |"
