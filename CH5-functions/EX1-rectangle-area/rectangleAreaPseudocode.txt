Module main()
    Declare Real width
    Declare Real length
    Declare Real area

    Output "Enter a width : "
    Input width
    Output "Enter a length : "
    Input length

    Set area = calculateArea(width, length)

    Output "Rectangle's area : ", area
End Module

Function Real calculateArea(Real width, Real length)
    Declare Real result

    Set result = width * length
    
    Return result
End Function
